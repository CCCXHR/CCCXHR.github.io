#include<iostream>

#include<stdlib.h>

{

	int n;
	int a1,a2,b1,b2,c1,c2;
	scanf("%d",&n);
	scanf("%d%d%d%d%d%d",&a1,&a2,&b1,&b2,&c1,&c2);
	int ans1=0,ans2=0,ans3=0;
	int ans;
	int ls=0;
	ls=n%a1;
	if(ls!=0)
		ans1=((n-ls)/a1+1)*a2;
	else 
		ans1=a1/n*a2;
	ls=n%b1;
	if(ls!=0)
		ans2=((n-ls)/b1+1)*b2;
	else 
		ans2=b1/n*b2;
	ls=n%c1;
	if(ls!=0)
		ans3=((n-ls)/c1+1)*c2;
	else 
		ans3=c1/n*c2;
	ans=min(min(ans1,ans2),ans3);
	printf("%d",ans); 
	return 0;
}

逐个查看要买多少和，比较得出最优解并输出。

#include<iostream>

#include<stdlib.h>

using namespace std;
inline bool check(int x)
{
	int pre=x;
	int q=0;
	while(x)
	{
		q=q*10+x%10;
		x/=10;
	}
	return (pre==q);
}
inline void hhh(int &val,int &date,int k)
{
	int t=1;
	for(register int i=0;i<k;++i)
	{
		val=val+(date%10)*t;
		date/=10;
		t*=10;
	}
}
inline bool rn(int year)
{
	return ((year%4==0&&year%100!=0)||(year%400==0));
}
inline int make(int y,int m,int d)
{
	return d+m*100+y*10000;
}
int month_day[13]={-1,31,28,31,30,31,30,31,31,30,31,30,31};
int main()
{
	int date1,date2;
	scanf("%d%d",&date1,&date2);

	{
		printf("%d",check(date1));
		return 0;
	}
	hhh(day,date1,2);
	hhh(month,date1,2);
	hhh(year,date1,4);
	hhh(td,date2,2);
	hhh(tm,date2,2);
	hhh(ty,date2,4);
	int ans=0;
	while(day!=td||month!=tm||year!=ty)
	{
// 		cout<<make(year,month,day)<<endl;

		if(check(make(year,month,day)))
			++ans;
		++day;

			md=(rn(year)==1)?29:28;
		if(day>md)
		{
			day=1;
			++month;
		}
		if(month==13)
		{
			month=1;
            ++year;
		}
	}
	if(check(make(year,month,day)))
		++ans;
	printf("%d",ans);
	return 0;
}

得分：70

思路：

记录每个乘客来到海港的时间、国籍、乘坐船只，同意加入数组（模拟队列）。对于每个乘客，对应国籍人数+1，乘客出队。每艘船统计完毕后输出答案并且清空结果数值。（这里需要逆序输出，所以是先压入了vector最后在输出的）

AC代码

#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<string.h>
#include<algorithm>
#include<string>
#include<queue>
using namespace std;
#define hour 86400
#define MAXX 100001
struct Pig
{
	int time,from;
};
queue<Pig> pig;
int cmp=1;
int tot[MAXX];
int main()
{
	int n,x,y,z;
	scanf("%d",&n);
	int ans=0;
	while(n--)
	{
		scanf("%d%d",&x,&y);
		cmp=pig.front().time;
		while(!pig.empty())
		{
			if(pig.front().time+hour>x)
				break;
			--tot[pig.front().from];
			if(tot[pig.front().from]==0)
				--ans;
			pig.pop();
		}
		while(y--)
		{
			scanf("%d",&z);
			pig.push((Pig){x,z});
			++tot[z];
			if(tot[z]==1)
				++ans;
		}
		printf("%d\n",ans);
	}
	return 0;
}

#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<algorithm>
using namespace std;
#define MAXM 2001
int ans[MAXM][4];
typedef struct node
{
	int val,index;
};
node x[MAXM];
bool cmp(node x,node y)
{
	return x.val<y.val;
}
int main()
{
//	freopen("magic.in","r",stdin);
//	freopen("magic.out","w",stdout);
	int n,m;
	scanf("%d%d",&n,&m);
	for(register int i=0;i<m;++i)
	{
		scanf("%d",&x[i].val);
		x[i].index=i;
	}
	sort(x,x+m,cmp);
	for(register int i=0;i<m;++i)
	{
		for(register int j=i+1;j<m;++j)
		{
			for(register int k=j+1;k<m;++k)
			{
				for(register int l=k+1;l<m;++l)
				{
					if((x[i].val<x[j].val&&x[j].val<x[k].val&&x[k].val<x[l].val)&&(x[j].val-x[i].val==2*(x[l].val-x[k].val))&&(x[j].val-x[i].val<(x[k].val*1.0-x[j].val*1.0)/3.0))
					{
//						cout<<x[i]<<" "<<x[j]<<" "<<x[k]<<" "<<x[l]<<endl;
						++ans[x[i].index][0];
						++ans[x[j].index][1];
						++ans[x[k].index][2];
						++ans[x[l].index][3];
					}
				}
			}
		}
	}
	for(register int i=0;i<m;++i)
	{
		for(register int j=0;j<4;++j)
			printf("%d ",ans[i][j]);
		printf("\n");
	}
	return 0;
}

因为

大魔法师认为，当且仅当四个编号为a,b,c,da,b,c,da,b,c,d的魔法物品满足 Xa<Xb<Xc<XdX_a < X_b < X_c < X_dXa​<Xb​<Xc​<Xd​ ······

可以对其进行一次快速排序，然后就可以缩小i,j,k,li,j,k,li,j,k,l四个变量的枚举范围。
